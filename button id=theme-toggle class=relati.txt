    <script>
        const toggleBtn = document.getElementById('theme-toggle');
        const html = document.documentElement;
        toggleBtn.addEventListener('click', () => {
            html.classList.toggle('dark');
            localStorage.setItem('theme', html.classList.contains('dark') ? 'dark' : 'light');
        });

        async function fetchProfile() {
            try {
                const response = await fetch('
https://primebasket.onrender.com/api/dashboard/
');
                const user = await response.json();
                const profileInfo = document.getElementById('profile-info');

                profileInfo.innerHTML = `
          <h3 class="text-xl font-semibold">${
user.name
 || 'No Name'}</h3>
          <p class="text-gray-500">Email: ${
user.email
 || 'No Email'}</p>
          <p class="text-gray-500">Phone: ${
user.phone
 || 'No Phone'}</p>
          <p class="text-gray-500">Joined: ${user.joined_date || 'Unknown'}</p>
          <p class="text-gray-500">Address: ${user.address || 'No Address'}</p>
        `;
            } catch (error) {
                console.error('Error fetching profile:', error);
                document.getElementById('profile-info').innerHTML = '<p class="text-center text-red-500">Failed to load profile. Please try again later.</p>';
            }
        }

        document.addEventListener('DOMContentLoaded', fetchProfile);
    </script>
